  # Simple return type refinement
  def always_positive(x: Int): {result: Int | result >= 0} =
    match x do
      n when n >= 0 -> n
      n -> 0 - n
    end
  
  # Return type with complex constraint
  def make_percentage(raw: Int): {p: Int | p >= 0 and p <= 100} =
    match raw do
      v when v < 0 -> 0
      v when v > 100 -> 100
      v -> v
    end

      x when x == -5 -> "-5"
